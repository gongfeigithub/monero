set(CMAKE_CXX_STANDARD 14)

set(gen_staking_client_sources
  misc.cpp
  gen_staking_client.cpp)

monero_add_executable(gen_staking_client
  ${gen_staking_client_sources})
target_link_libraries(gen_staking_client
  PRIVATE
    wallet
    cryptonote_core
    cncrypto
    common
    epee
    ${EPEE_READLINE}
    ${Boost_CHRONO_LIBRARY}
    ${Boost_PROGRAM_OPTIONS_LIBRARY}
    ${Boost_FILESYSTEM_LIBRARY}
    ${Boost_THREAD_LIBRARY}
    ${CMAKE_THREAD_LIBS_INIT}
    ${GNU_READLINE_LIBRARY}
    ${EXTRA_LIBRARIES})
add_dependencies(gen_staking_client
  version)
set_property(TARGET gen_staking_client
  PROPERTY
    OUTPUT_NAME "mendark-gen-staking-client")
install(TARGETS gen_staking_client DESTINATION bin)
